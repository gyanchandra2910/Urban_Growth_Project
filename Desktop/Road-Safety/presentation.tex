\begin{frame}{Problem Statement}
\framesubtitle{The Challenge in Road Safety Management}

\begin{block}{Current Scenario}
    \begin{itemize}
        \item India accounts for \textbf{11\% of global road accident deaths} despite having only 1\% of world's vehicles
        \item Over \textbf{150,000 fatalities annually} on Indian roads
        \item Identifying appropriate safety interventions is \textbf{time-consuming and expertise-dependent}
    \end{itemize}
\end{block}

\vspace{0.3cm}

\begin{block}{Key Challenges}
    \begin{itemize}
        \item \textbf{Manual Analysis:} Engineers spend hours matching problems to IRC-compliant solutions
        \item \textbf{Inconsistent Decisions:} Different experts may recommend different interventions
        \item \textbf{Knowledge Gap:} Limited access to comprehensive intervention databases
        \item \textbf{Delayed Response:} Slow identification leads to preventable accidents
    \end{itemize}
\end{block}

\vspace{0.2cm}
\begin{center}
    \textcolor{secondaryblue}{\textit{"How can we automate and optimize road safety intervention recommendations?"}}
\end{center}

\end{frame}

    \node[box, fill=orange!20, below=1cm of gpt] (output) {OUTPUT\\{\small Ranked Interventions\\+ AI Explanation}};

    \draw[arrow] (tfidf) -- node[above] {\small Top 5} (gpt);
    \draw[arrow] (gpt) -- (output);
    \draw[arrow] (db) -- (tfidf);
    \draw[arrow] (db) -- (gpt);
\end{tikzpicture}
\end{center}

\vspace{0.3cm}

\begin{block}{Workflow Steps}
    \begin{enumerate}
        \item \textbf{Step 1:} User describes road safety issue (e.g., "damaged signs at highway exit")
        \item \textbf{Step 2:} TF-IDF vectorizer searches database, ranks by cosine similarity
        \item \textbf{Step 3:} Top matches sent to GPT-3.5 with context (problem type, IRC clauses, details)
        \item \textbf{Step 4:} GPT generates professional explanation referencing specific standards
        \item \textbf{Step 5:} System returns ranked interventions with relevance scores + AI insight
    \end{enumerate}
\end{block}

\end{frame}

\begin{frame}{Backup: Technical Details}
\framesubtitle{System Specifications}

\begin{block}{TF-IDF Configuration}
    \begin{itemize}
        \item \textbf{Max Features:} 1000 terms
        \item \textbf{N-gram Range:} 1-2 (unigrams + bigrams)
        \item \textbf{Stop Words:} English corpus
        \item \textbf{Similarity Metric:} Cosine similarity
    \end{itemize}
\end{block}

\begin{block}{GPT Integration}
    \begin{itemize}
        \item \textbf{Model:} GPT-3.5-turbo
        \item \textbf{Temperature:} 0.7 (balanced creativity)
        \item \textbf{Max Tokens:} 250 per explanation
        \item \textbf{Context Window:} Top 3 matches + IRC references
    \end{itemize}
\end{block}

\begin{block}{Deployment}
    \begin{itemize}
        \item \textbf{Backend:} Flask on port 5000
        \item \textbf{Frontend:} Streamlit on port 8501
        \item \textbf{Launcher:} Python script with health checks
        \item \textbf{Platform:} Cross-platform (Windows/Linux/Mac)
    \end{itemize}
\end{block}

\end{frame}
